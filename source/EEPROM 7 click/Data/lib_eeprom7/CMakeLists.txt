cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_eeprom7 LANGUAGES MikroC)
else()
    project(lib_eeprom7 LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_eeprom7 STATIC
        src/eeprom7.c
        include/eeprom7.h
)
add_library(Click.EEPROM7  ALIAS lib_eeprom7)



target_include_directories(lib_eeprom7 PUBLIC
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_eeprom7 PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_eeprom7 PUBLIC MikroSDK.Driver)
